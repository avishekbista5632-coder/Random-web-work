// Program to determine the type of triangle based on its sides function triangleType(side1, side2, side3) { if (side1 === side2 && side2 === side3) { return "Equilateral triangle"; } else if (side1 === side2 || side2 === side3 || side1 === side3) { return "Isosceles triangle"; } else { return "Scalene triangle"; } } // Example usage: let side1 = parseFloat(prompt("Enter the length of side 1:")); let side2 = parseFloat(prompt("Enter the length of side 2:")); let side3 = parseFloat(prompt("Enter the length of side 3:")); console.log("Type of triangle:", triangleType(side1, side2, side3));
